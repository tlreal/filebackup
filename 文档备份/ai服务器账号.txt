class ChatRequest(BaseModel):
    prompt: str
    image_base64: Optional[str] = None
    max_tokens: int = 512
    temperature: float = 0.8

class ChatMessage(BaseModel):
    role: str
    content: Union[str, List[Any]]

class ChatCompletionRequest(BaseModel):
    model: Optional[str] = "glm-4v-9b"
    messages: List[ChatMessage]
    max_tokens: Optional[int] = 512
    temperature: Optional[float] = 0.7
    top_p: Optional[float] = 0.9
    stream: Optional[bool] = False

class ResponsesRequest(BaseModel):
    model: Optional[str] = "glm-4v-9b"
    input: Any
    max_tokens: Optional[int] = 512
    temperature: Optional[float] = 0.7
    top_p: Optional[float] = 0.9

def process_base64(b64_str: str) -> str:
    """清理并修复 Base64 字符串"""
    b64_str = b64_str.strip()
    if "," in b64_str:
        b64_str = b64_str.split(",")[1]
    
    # 修复 Padding
    missing_padding = len(b64_str) % 4
    if missing_padding:
        b64_str += '=' * (4 - missing_padding)
    
    # 验证是否合法
    try:
        base64.b64decode(b64_str)
    except Exception:
        raise ValueError("Invalid Base64 string")
        
    return b64_str

@app.post("/chat")
async def chat(request: ChatRequest):
    try:
        image_obj = None
        if request.image_base64:
            try:
                clean_b64 = process_base64(request.image_base64)
                img_bytes = base64.b64decode(clean_b64)
                image_obj = Image.open(io.BytesIO(img_bytes)).convert("RGB")
            except Exception as img_err:
                raise HTTPException(status_code=400, detail=f"图片解析失败: {str(img_err)}")

        if image_obj is None:
            image_obj = Image.new("RGB", (224, 224), color="black")

        inputs = tokenizer.apply_chat_template(
            [
                {
                    "role": "user",
                    "image": image_obj,
                    "content": request.prompt,
                }
            ],
            add_generation_prompt=True,
            tokenize=True,
            return_tensors="pt",
            return_dict=True,
        )

        device = next(model.parameters()).device
        inputs = inputs.to(device)

        with torch.no_grad():
            outputs = model.generate(
                **inputs,
                max_new_tokens=request.max_tokens,
                do_sample=True if request.temperature > 0 else False,
                temperature=request.temperature,
                top_p=0.8,
                repetition_penalty=1.1,
            )

        input_len = inputs["input_ids"].shape[1]
        output_text = tokenizer.decode(outputs[0][input_len:], skip_special_tokens=True)

        return {"response": output_text.strip()}

    except HTTPException:
        raise
    except Exception as e:
        print(f"❌ 处理请求时出错: {e}")
        import traceback
        traceback.print_exc()
        raise HTTPException(status_code=500, detail=str(e))

@app.post("/v1/chat/completions")
async def chat_completions(request: ChatCompletionRequest):
    if model is None or tokenizer is None:
        raise HTTPException(status_code=503, detail="模型正在加载中，请稍后再试")
    try:
        if not request.messages:
            raise HTTPException(status_code=400, detail="messages 不能为空")
        last_message = request.messages[-1]
        if last_message.role != "user":
            for msg in reversed(request.messages):
                if msg.role == "user":
                    last_message = msg
                    break
        prompt_text = ""
        image_obj = None
        if isinstance(last_message.content, str):
            prompt_text = last_message.content
        elif isinstance(last_message.content, list):
            for item in last_message.content:
                if isinstance(item, dict):
                    item_type = item.get("type")
                    if item_type == "text":
                        prompt_text += str(item.get("text", ""))
                    elif item_type == "image_url":
                        image_url = item.get("image_url")
                        url = None
                        if isinstance(image_url, dict):
                            url = image_url.get("url")
                        elif isinstance(image_url, str):
                            url = image_url
                        if not url:
                            continue
                        if "," in url:
                            b64_part = url.split(",", 1)[1]
                        else:
                            b64_part = url
                        clean_b64 = process_base64(b64_part)
                        img_bytes = base64.b64decode(clean_b64)
                        image_obj = Image.open(io.BytesIO(img_bytes)).convert("RGB")
        if image_obj is None:
            image_obj = Image.new("RGB", (224, 224), color="black")
            if not prompt_text:
                prompt_text = "描述这张图片"
        inputs = tokenizer.apply_chat_template(
            [
                {
                    "role": "user",
                    "image": image_obj,
                    "content": prompt_text,
                }
            ],
            add_generation_prompt=True,
            tokenize=True,
            return_tensors="pt",
            return_dict=True,
        )
        device = next(model.parameters()).device
        inputs = inputs.to(device)
        max_tokens = request.max_tokens if request.max_tokens is not None else 512
        temperature = request.temperature if request.temperature is not None else 0.7
        top_p = request.top_p if request.top_p is not None else 0.9
        with torch.no_grad():
            outputs = model.generate(
                **inputs,
                max_new_tokens=max_tokens,
                do_sample=True if temperature > 0 else False,
                temperature=temperature,
                top_p=top_p,
                repetition_penalty=1.1,
            )
        input_len = inputs["input_ids"].shape[1]
        output_tokens = outputs[0]
        completion_tokens = output_tokens[input_len:]
        output_text = tokenizer.decode(completion_tokens, skip_special_tokens=True).strip()
        prompt_tokens_count = int(input_len)
        completion_tokens_count = int(completion_tokens.shape[0])
        total_tokens_count = int(output_tokens.shape[0])
        now = int(time.time())
        response = {
            "id": "chatcmpl-" + str(now),
            "object": "chat.completion",
            "created": now,
            "model": request.model or "glm-4v-9b",
            "choices": [
                {
                    "index": 0,
                    "message": {
                        "role": "assistant",
                        "content": output_text,
                    },
                    "finish_reason": "stop",
                }
            ],
            "usage": {
                "prompt_tokens": prompt_tokens_count,
                "completion_tokens": completion_tokens_count,
                "total_tokens": total_tokens_count,
            },
        }
        return response
    except HTTPException:
        raise
    except Exception as e:
        print(f"❌ 处理 OpenAI 请求时出错: {e}")
        import traceback
        traceback.print_exc()
        raise HTTPException(status_code=500, detail=str(e))
